function formatTime(sec){
  if(isNaN(sec)) return "0:00";
  let minutes = Math.floor(sec / 60);
  let seconds = Math.floor(sec % 60);
  return minutes + ":" + (seconds < 10 ? "0" : "") + seconds;
}
